## Overview

TradeMaster supports multiple payment methods including international credit cards via Stripe, domestic Venezuelan payment methods via Mercantil Bank API, and BCV mobile payments. The system provides a unified interface while handling the complexities of each payment gateway.

## Stripe Integration

### Configuration

Stripe is initialized using environment variables:

```javascript title="src/guards/stripe/stripeConfig.js"
import { loadStripe } from '@stripe/stripe-js';

const STRIPE_PUBLIC_KEY = import.meta.env.VITE_STRIPE_PUBLIC_KEY || 'pk_test_...';

let stripePromise = null;

export const initializeStripe = async () => {
  if (!STRIPE_PUBLIC_KEY || STRIPE_PUBLIC_KEY === 'pk_test_...') {
    console.error('❌ STRIPE_PUBLIC_KEY no está configurada.');
    throw new Error('La clave pública de Stripe no está configurada');
  }

  if (!stripePromise) {
    stripePromise = loadStripe(STRIPE_PUBLIC_KEY);
  }
  return stripePromise;
};

export const getStripePromise = () => stripePromise;
export default stripePromise;
```

### Payment Intent Flow

<Steps>
  <Step title="Create Payment Intent">
    Create a payment intent on the server with the order amount:
    
    ```javascript
    const createPaymentIntent = async (amount, currency = 'usd') => {
      try {
        const response = await stripe_create_payment_intent({
          amount: Math.round(amount * 100), // Convert to cents
          currency: currency,
          metadata: {
            event_id: eventId,
            user_id: userId,
            ticket_count: selectedSeats.length
          }
        });
        
        setStripeClientSecret(response.data.clientSecret);
        setStripePaymentIntentId(response.data.paymentIntentId);
      } catch (error) {
        console.error('Error creating payment intent:', error);
      }
    };
    ```
  </Step>
  
  <Step title="Collect Payment Details">
    Use Stripe Elements to securely collect card information:
    
    ```javascript title="src/views/purchaseOrder/PurchaseOrder.js:36-83"
    const StripeCardInput = ({ onCardElementReady }) => {
      const stripe = useStripe();

          onCardElementReady(cardElement);
        }
      }, [stripe, elements, onCardElementReady]);

      return (
        <div className="grid gap-2">
          <Label>Información de la Tarjeta</Label>
          <div className="p-4 border rounded-md">
            <CardElement
              options={{
                style: {
                  base: {
                    fontSize: '16px',
                    color: 'hsl(var(--foreground))',
                    fontFamily: '"Inter", sans-serif',
                    '::placeholder': {
                      color: 'hsl(var(--muted-foreground))',
                    },
                  },
                  invalid: {
                    color: 'hsl(var(--destructive))',
                  },
                },
              }}
            />
          </div>
        </div>
      );
    };
    ```
  </Step>
  
  <Step title="Confirm Payment">
    Confirm the payment with Stripe:
    
    ```javascript
    const confirmStripePayment = async (cardElement) => {
      try {
        setStripeProcessing(true);
        
        const { error, paymentIntent } = await stripe.confirmCardPayment(
          stripeClientSecret,
          {
            payment_method: {
              card: cardElement,
              billing_details: {
                name: stripeInternationalInfo.cardholder_name,
                email: stripeInternationalInfo.email,
              },
            },
          }
        );

        if (error) {
          throw new Error(error.message);
        }

        if (paymentIntent.status === 'succeeded') {
          // Payment successful - create order
          await createOrder(paymentIntent.id);
        }
      } catch (error) {
        console.error('Error confirming payment:', error);
        toast.error('Error processing payment');
      } finally {
        setStripeProcessing(false);
      }
    };
    ```
  </Step>
  
  <Step title="Create Order">
    Once payment is confirmed, create the order and assign tickets:
    
    ```javascript
    const createOrder = async (paymentIntentId) => {
      const orderData = {
        event_id: eventId,
        user_id: userId,
        tickets: selectedSeats.map(seat => seat.ticket_id),
        payment_method: 'Tarjeta de Crédito Internacional',
        payment_intent_id: paymentIntentId,
        amount: totalAmount,
        status: 'completed'
      };
      
      await order_created(orderData);
      navigate(`/purchaseSuccess/${eventId}/${orderId}`);
    };
    ```
  </Step>
</Steps>

### Debit Card with 2FA

Debit cards require two-factor authentication:

```javascript title="src/views/purchaseOrder/PurchaseOrder.js:568-654"
const authenticateDebitCard = async () => {
  if (!debitCardInfo.card_number || !debitCardInfo.identification_number) {
    toast.error('Ingrese el número de tarjeta y la identificación');
    return;
  }

  try {
    setDebitCardAuth(prev => ({ 
      ...prev, 
      authInProgress: true, 
      authError: null 
    }));

    // Step 1: Authenticate the card
    const authResponse = await api_mercantil_auth({
      customer_id: `${debitCardInfo.identification_type}${debitCardInfo.identification_number}`,
      card_number: debitCardInfo.card_number,
      ipaddress: ipAddress,
      browser_agent: navigator.userAgent,
      manufacturer: null
    });

    if (authResponse?.data?.authentication_info.trx_status === "approved") {
      setDebitCardAuth(prev => ({
        ...prev,
        isAuthenticated: true,
        authInProgress: false,
        authData: authResponse.data
      }));

      toast.success('Tarjeta autenticada correctamente');
    } else {
      throw new Error(authResponse?.data?.message || 'Authentication failed');
    }
  } catch (error) {
    setDebitCardAuth(prev => ({
      ...prev,
      isAuthenticated: false,
      authInProgress: false,
      authError: error.message
    }));

    toast.error('Error al autenticar la tarjeta');
  }
};

// Step 2: Process the payment after authentication
const processDebitCardPayment = async () => {
  if (!debitCardAuth.isAuthenticated) {
    toast.error('Por favor autentique la tarjeta primero');
    return;
  }

  try {
    const response = await api_mercantil_tdd_pay({
      authentication_data: debitCardAuth.authData,
      card_number: debitCardInfo.card_number,
      expiry_date: debitCardInfo.expiry_date,
      cvv: debitCardInfo.cvv,
      twofactor_auth: debitCardInfo.twofactor_auth,
      account_type: debitCardInfo.account_type,
      amount: totalAmount,
      currency: 'VES'
    });

    if (response.data.status === 'approved') {
      await createOrder(response.data.transaction_id);
    }
  } catch (error) {
    console.error('Error processing debit card:', error);
  }
};
```

## Order Creation

Once payment is confirmed, create the order:

```javascript title="Order Creation Flow"
const createOrder = async (transactionId) => {
  try {
    const orderData = {
      event_id: eventDetails.id,
      client_id: eventDetails.client_id,
      user_id: user.id,
      buyer_info: buyerInfo,
      recipient_info: isGiftPurchase ? recipientInfo : buyerInfo,
      tickets: selectedSeats.map(seat => ({
        ticket_id: seat.ticket_id,
        seat_id: seat.id,
        zone: seat.location,
        price: seat.price
      })),
      payment_method: paymentMethod,
      payment_transaction_id: transactionId,
      amount: totalAmount,
      currency: paymentMethod.includes('Internacional') ? 'USD' : 'VES',
      status: 'completed',
      created_at: new Date().toISOString()
    };

    const response = await order_created(orderData);

    if (response.data.success) {
      // Send confirmation email
      await send_email({
        to: buyerInfo.email,
        template: 'order_confirmation',
        data: {
          order_id: response.data.order_id,
          event_name: eventDetails.title,
          tickets: selectedSeats
        }
      });

      navigate(`/purchaseSuccess/${eventDetails.id}/${response.data.order_id}`);
    }
  } catch (error) {
    console.error('Error creating order:', error);
    toast.error('Error al crear la orden');
  }
};
```

## Exchange Rate Handling

For Venezuelan bolivar payments, convert USD to VES:

```javascript title="Exchange Rate Conversion"
const usdCentralBank = useSelector(
  (state) => state.configData?.config_data?.exchange_rates?.rates?.usd ?? 0
);

// Helper function to truncate decimals
const truncateDecimals = (number, digits) => {
  const multiplier = Math.pow(10, digits);
  return Math.trunc(number * multiplier) / multiplier;
};

// Convert amount to VES
const amountInVES = truncateDecimals(totalAmount * usdCentralBank, 2);
```

## Payment Validation

```javascript title="src/views/purchaseOrder/PurchaseOrder.js:656-714"
const isPaymentInfoValid = () => {
  switch (paymentMethod) {
    case "Pago Móvil":
      const selectedMethod = paymentMethodsData.find(
        p => p.method === paymentMethod
      );
      const bankName = selectedMethod?.data?.bank || "";

      if (!pagoMovilInfo.origin_bank || !pagoMovilInfo.phone) {
        return false;
      }

      if (bankName === "Mercantil") {
        return pagoMovilInfo.twofactor_auth !== "";
      }

      if (bankName === "Banco de Venezuela") {
        return pagoMovilInfo.pagador_id_type &&
               pagoMovilInfo.pagador_id &&
               pagoMovilInfo.reference_number &&
               pagoMovilInfo.reference_number.length >= 8;
      }
      return true;

    case "Tarjeta de Crédito":
      return creditCardInfo.card_number !== "" &&
             creditCardInfo.expiry_date !== "" &&
             creditCardInfo.cvv !== "" &&
             creditCardInfo.cardholder_name !== "" &&
             creditCardInfo.identification_number !== "";

    case "Tarjeta de Debito":
      return debitCardAuth.isAuthenticated &&
             debitCardInfo.cardholder_name !== "" &&
             debitCardInfo.expiry_date !== "" &&
             debitCardInfo.cvv !== "" &&
             debitCardInfo.twofactor_auth !== "";

    case "Tarjeta de Crédito Internacional":
      return stripeInternationalInfo.cardholder_name !== "" &&
             stripeInternationalInfo.email !== "";

    default:
      return false;
  }
};
```

## Best Practices

1. **Never store sensitive card data** - Use tokenization for all card processing
2. **Implement proper error handling** - Show user-friendly error messages
3. **Validate payment info** before attempting to process
4. **Use HTTPS** for all payment-related requests
5. **Log all transactions** for audit and reconciliation purposes
6. **Handle currency conversion** carefully to avoid rounding errors
7. **Implement timeout handling** to release locked tickets if payment fails
8. **Send confirmation emails** after successful payment
9. **Store transaction IDs** for future reference and dispute resolution
10. **Test with different payment methods** to ensure compatibility
