  <Step title="Apply Filters on Change">
    Re-filter events whenever any filter changes:
    
    ```javascript title="src/views/eventList/EventList.js:146-216"
    useEffect(() => {
      const filterEvents = () => {
        let filtered = [...events];
        
        // Apply search filter
        if (searchTerm) {
          filtered = filtered.filter(event =>
            event.title.toLowerCase().includes(searchTerm.toLowerCase())
          );
        }
        
        // Apply category filter
        if (selectedCategory !== 'Todos') {
          filtered = filtered.filter(event => 
            event.genre === selectedCategory
          );
        }
        
        // Apply location filter
        if (selectedLocation !== 'Todas') {
          filtered = filtered.filter(event => 
            event.location === selectedLocation
          );
        }
        
        // Apply month filter
        if (selectedMonth !== 'Todos') {
          filtered = filtered.filter(event => 
            event.month === Number(selectedMonth)
          );
        }
        
        // Apply year filter
        if (selectedYear !== 'Todos') {
          filtered = filtered.filter(event => 
            event.year === Number(selectedYear)
          );
        }
        
        // Sort by date
        filtered = filtered.sort((a, b) => a.dateObj - b.dateObj);
        
        setFilteredEvents(filtered);
        setDisplayedEvents(filtered.slice(0, 9));
      };
      
      filterEvents();
    }, [events, searchTerm, selectedCategory, selectedLocation, selectedMonth, selectedYear]);
    ```
  </Step>

## Event Detail Page

The event detail view provides comprehensive information about a single event, including pricing, zones, venue details, and related events.

### Loading Event Data

```javascript title="src/views/event/Event.js:22-83"
useEffect(() => {
  const getEventData = () => {
    dispatch(setLoadingData({ loading: true }));
    window.scrollTo(0, 0);
    
    const actualEvent = events.find((x) => x.event_id === params.id);
    
    const actualEventData = actualEvent ? {
      title: actualEvent.event_name,
      dateStart: convertTimestampLL(actualEvent.date_start),
      dateEnd: convertTimestampLL(actualEvent.date_end),
      timeStart: convertTimestampToHour(actualEvent.date_start),
      timeEnd: convertTimestampToHour(actualEvent.date_end),
      venue: actualEvent.venue_name,
      venueId: actualEvent.venue_id,
      description: actualEvent.event_desc,
      image: actualEvent.event_media[0],
      maxCapacity: actualEvent.venue_max_capacity,
      eventType: actualEvent.type,
      status: actualEvent.status,
      zones: actualEvent.zones.zone.map((zone) => {
        const basePrice = zone.price;
        const tax = basePrice * 0.16;
        const totalPrice = basePrice + tax;
        return {
          id: zone.id,
          name: zone.name,
          type: zone.type,
          description: zone.description,
          seats: zone.seats,
          color: zone.color,
          price: basePrice,
          tax: tax,
          total: totalPrice,
        }
      }),
    } : {};
    
    setEventDetails(actualEventData);
    dispatch(setLoadingData({ loading: false }));
  }
  
  getEventData();
}, [params.id, events]);
```

## Best Practices

1. **Always filter by tenant** - Events should be filtered by `client_id` to ensure tenant isolation
2. **Sort by date** - Display events in chronological order by default
3. **Show active events only** - Filter out inactive events (`status === 'Activo'`)
4. **Calculate prices with tax** - Always display final prices including 16% IVA
5. **Provide multiple filter options** - Allow users to filter by category, location, and date
6. **Implement pagination** - Load events in batches for better performance
