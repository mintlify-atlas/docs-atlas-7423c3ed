## Overview

The Events module consists of two main views:
- **Event Detail** (`src/views/event/Event.js`) - Individual event information page
- **Event List** (`src/views/eventList/EventList.js`) - Searchable and filterable events catalog

## Event Detail View

**Route:** `/event/:id`

### Purpose

Displays comprehensive information about a specific event including dates, venue, pricing zones, and booking options.

### Redux Integration

```javascript
const events = useSelector((state) => state.events.events);
```

### State Management


<Steps>
  <Step title="Extract event ID from URL">

  <Step title="Find event in Redux store">

### Dynamic Filter Extraction

```javascript
useEffect(() => {
  const loadEvents = async () => {
    dispatch(setLoadingData({ loading: true }));

    const uniqueLocations = ['Todas'];
    const months = new Set(['Todos']);
    const years = new Set(['Todos']);

    events.forEach(evento => {
      if (evento?.venue_name && !uniqueLocations.includes(evento.venue_name)) {
        uniqueLocations.push(evento.venue_name);
      }

      if (evento?.date_start) {
        const dateStr = convertTimestamp(evento.date_start);
        const date = parseDate(dateStr);
        months.add(date.getMonth());
        years.add(date.getFullYear());
      }
    });

    setLocations(uniqueLocations);
    setAvailableMonths([...months].sort());
    setAvailableYears([...years].sort());
    
    // Process and sort events
    const processedEvents = events.map((evento) => ({
      id: evento?.event_id,
      title: evento?.event_name,
      date: convertTimestamp(evento?.date_start),
      dateObj: parseDate(convertTimestamp(evento?.date_start)),
      month: parseDate(convertTimestamp(evento?.date_start)).getMonth(),
      year: parseDate(convertTimestamp(evento?.date_start)).getFullYear(),
      location: evento?.venue_name,
      image: evento.event_media[0] !== '' ? evento.event_media[0] : "https://placehold.co/1200x600/png",
      genre: evento?.type || 'Otros',
      description: evento?.event_desc,
    })).sort((a, b) => a.dateObj - b.dateObj);

    setFilteredEvents(processedEvents);
    setDisplayedEvents(processedEvents.slice(0, 9));
    dispatch(setLoadingData({ loading: false }));
  };

  loadEvents();
}, [events, dispatch]);
```

### Search and Filter Logic

```javascript
useEffect(() => {
  const filterEvents = () => {
    let filtered = [...events].map((evento) => ({
      id: evento?.event_id,
      title: evento?.event_name,
      date: convertTimestamp(evento?.date_start),
      dateObj: parseDate(convertTimestamp(evento?.date_start)),
      month: parseDate(convertTimestamp(evento?.date_start)).getMonth(),
      year: parseDate(convertTimestamp(evento?.date_start)).getFullYear(),
      location: evento?.venue_name,
      image: evento.event_media[0],
      genre: evento?.type || 'Otros',
      description: evento?.event_desc,
    }));

    // Filter by search term
    if (searchTerm) {
      filtered = filtered.filter(
        event =>
          event.title.toLowerCase().includes(searchTerm.toLowerCase()) ||
          event.description?.toLowerCase().includes(searchTerm.toLowerCase()) ||
          event.location.toLowerCase().includes(searchTerm.toLowerCase())
      );
    }

    // Filter by category
    if (selectedCategory !== 'Todos') {
      filtered = filtered.filter(event => event.genre === selectedCategory);
    }

    // Filter by location
    if (selectedLocation !== 'Todas') {
      filtered = filtered.filter(event => event.location === selectedLocation);
    }

    // Filter by month
    if (selectedMonth !== 'Todos') {
      filtered = filtered.filter(event => event.month === Number(selectedMonth));
    }

    // Filter by year
    if (selectedYear !== 'Todos') {
      filtered = filtered.filter(event => event.year === Number(selectedYear));
    }

    // Always sort by date
    filtered = filtered.sort((a, b) => a.dateObj - b.dateObj);

    setFilteredEvents(filtered);
    setDisplayedEvents(filtered.slice(0, 9));
  };

  filterEvents();
}, [events, searchTerm, selectedCategory, selectedLocation, selectedMonth, selectedYear]);
```
