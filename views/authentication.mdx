## Overview

The authentication module provides a complete user authentication system with four main views:

<CardGroup cols={2}>
  <Card title="Email Check" icon="envelope">
    Initial email verification step
  </Card>
  <Card title="Login" icon="lock">
    User authentication with credentials
  </Card>
  <Card title="Register" icon="user-plus">
    New account creation
  </Card>
  <Card title="Forgot Password" icon="key">
    Password reset functionality
  </Card>
</CardGroup>

**Base Route:** `/auth`  
**Layout:** `<BlankLayout>` with `<GuestGuard>`

## Authentication Flow

<Steps>
  <Step title="Email Verification">
    User enters email at `/auth/checkEmail`
  </Step>
  <Step title="Route Decision">
    System checks if account exists and is validated
  </Step>
  <Step title="Login or Register">
    Redirect to `/auth/login` (existing) or `/auth/register` (new/incomplete)
  </Step>
  <Step title="Access Granted">
    Successful authentication updates Redux store and grants access
  </Step>
</Steps>

## Email Check View

**Route:** `/auth/checkEmail` (default `/auth` route)

### Purpose

First step in authentication flow that determines whether to direct users to login or registration.

### Implementation

```javascript
import { getDocs, query, collection, where } from "firebase/firestore";
import { setInitialEmail, setUserToComplete } from "../../../store/apps/auth/authSlice";

const EmailCheck = () => {
  const [submit, setSubmit] = useState(false);
  const dispatch = useDispatch();
  const navigate = useNavigate();

  const LoginSchema = Yup.object().shape({
    email: Yup.string()
      .email("El correo electrónico no es válido")
      .required("Se requiere correo electrónico"),
  });

  const formik = useFormik({
    initialValues: {
      email: "",
    },
    validationSchema: LoginSchema,
    onSubmit: async (values, { setSubmitting }) => {
      try {
        setSubmit(true);
        const q = query(collection(db, "u_users"), where("email", "==", values.email.toLowerCase().trim()));
        const querySnapshot = await getDocs(q);

        if(querySnapshot.docs.length === 0){
          toast.error("No existe una cuenta asociada a este correo. Regístrate para proceder.");
          setSubmitting(false);
          setSubmit(false);
          return;
        } else {
          if (querySnapshot.docs[0].data().account_validated) {
            dispatch(setInitialEmail({email: values.email.toLowerCase().trim()}));
            navigate('/auth/login');
          } else {
            dispatch(setUserToComplete({user: {...querySnapshot.docs[0].data(), uid: querySnapshot.docs[0].id}}));
            navigate('/auth/register');
          }
        }
      } catch (error) {
        console.error("Email check error:", error);
        toast.error("Se produjo un error. Inténtelo de nuevo.");
        setSubmitting(false);
        setSubmit(false);
      }
    },
  });

  return (
    <div className="min-h-screen bg-gradient-to-b from-primary/10 to-secondary/10 flex items-center justify-center">
      <Card className="w-full max-w-md">
        <CardHeader>
          <CardTitle className="text-2xl font-bold text-center">Verificar Correo</CardTitle>
          <CardDescription className="text-center">
            Ingresa tu correo electrónico para continuar
          </CardDescription>
        </CardHeader>
        <FormikProvider value={formik}>
          <Form onSubmit={formik.handleSubmit}>
            <CardContent>
              <div className="space-y-2">
                <Label htmlFor="email">Correo electrónico</Label>
                <div className="relative">
                  <Input
                    id="email"
                    placeholder="tu@ejemplo.com"
                    type="email"
                    className="pl-10"
                    {...formik.getFieldProps("email")}
                  />
                  <MailIcon className="absolute left-3 top-1/2 transform -translate-y-1/2 text-gray-400 h-5 w-5" />
                </div>
                {formik.touched.email && formik.errors.email && (
                  <Alert variant="destructive">
                    <AlertCircle className="h-4 w-4 mr-2" />
                    {formik.errors.email}
                  </Alert>
                )}
              </div>
            </CardContent>
            <CardFooter className="flex flex-col space-y-4">
              <Button type="submit" className="w-full bg-primary hover:bg-primary/90 text-white" disabled={submit}>
                {submit ? 'Verificando...' : 'Iniciar sesión'}
              </Button>
            </CardFooter>
          </Form>
        </FormikProvider>
      </Card>
    </div>
  );
};
```

### Flow Logic

1. **No account exists**: Show error message prompting registration
2. **Account exists and validated**: Store email in Redux and navigate to login
3. **Account exists but not validated**: Store user data in Redux and navigate to registration to complete

## Login View

**Route:** `/auth/login`

### Purpose

Authenticate existing users with email and password.

### Redux Integration


### Form Implementation

```javascript
const LoginSchema = Yup.object().shape({
  email: Yup.string()
    .email("El correo electrónico no es válido")
    .required("Se requiere correo electrónico"),
  password: Yup.string()
    .min(6, "La contraseña debe tener al menos 6 caracteres.")
    .required("Se requiere contraseña"),
});

const formik = useFormik({
  initialValues: {
    email: initialEmail ?? "",
    password: "",
  },
  validationSchema: LoginSchema,
  onSubmit: async (values, { setStatus, setSubmitting }) => {
    try {
      setSubmit(true);
      
      const userCredential = await signInWithEmailAndPassword(
        auth,
        values.email,
        values.password
      );

      const userUid = userCredential.user.uid;
      const querySnapshot = await getDoc(doc(db, "u_users", userUid));

      if (querySnapshot.data()?.status !== true) {
        await signOut(auth);
        setSubmit(false);
        
        dispatch(
          authStateChanged({
            isAuthenticated: false,
            user: null,
          })
        );
      } else {
        dispatch(
          authStateChanged({
            isAuthenticated: true,
            user: {
              id: userUid,
              email: querySnapshot.data()?.email,
              name: querySnapshot.data()?.name,
              ...querySnapshot.data(),
            },
          })
        );
        dispatch(
          setInitialEmail({
            email: null,
          })
        );
      }
    } catch (error) {
      console.error("Login error:", error);
      toast.error(getErrorMessage(error.code));
      setSubmitting(false);
      setSubmit(false);
    }
  },
});
```

### Error Handling

```javascript
const getErrorMessage = (errorCode) => {
  switch (errorCode) {
    case "auth/invalid-email":
      return "La dirección de correo electrónico no es válida.";
    case "auth/invalid-login-credentials":
      return "La contraseña es incorrecta.";
    case "auth/invalid-credential":
      return "La contraseña o el correo electrónico son incorrectos.";
    case "auth/too-many-requests":
      return "Has realizado muchas solicitudes, espera un momento y vuelve a intentarlo.";
    default:
      return "Se produjo un error al intentar iniciar sesión. Inténtelo de nuevo.";
  }
};
```

### Registration Logic

```javascript
onSubmit: async (values, { setStatus, setSubmitting }) => {
  try {
    setSubmit(true);
    
    if (userToComplete) {
      // Complete existing account
      const q = query(collection(db, `u_users/${userToComplete.uid}/security`));
      const securityQuery = await getDocs(q);
      let oldPass = securityQuery.docs[0].data().pin;

      const log = await signInWithEmailAndPassword(auth, userToComplete.email, oldPass);
      const oldUser = log.user;
      await updatePassword(oldUser, values.password.trim());

      const userData = {
        name: `${values.name.trim()} ${values.lastname.trim()}`,
        email: values.email,
        id: values.id,
        id_type: values.identification_type.value,
        genre: values.genre,
        phone: values.phone,
        status: true,
        uid: oldUser.uid,
        account_validated: true,
        "date.updated": serverTimestamp(),
        "date.validated": serverTimestamp(),
        address: {
          country: values.country,
          state: values.state
        },
      };

      await updateDoc(doc(db, "u_users", oldUser.uid), userData);
      dispatch(setInitialEmail({email: values.email}));
      dispatch(setUserToComplete({user: null}));
      toast.success("Registro completado exitosamente.");
      navigate('/auth/login');
    } else {
      // Create new account
      const userRegister = await createUserWithEmailAndPassword(
        auth,
        values.email.trim().toLowerCase(),
        values.password.trim()
      );
      
      const userUid = userRegister.user.uid;
      const finalData = {
        account_validated: false,
        address: {
          country: values.country,
          state: values.state
        },
        date: {
          created: serverTimestamp(),
          updated: '',
          validated: ''
        },
        email: values.email,
        genre: values.genre,
        id: values.id,
        id_type: values.identification_type.value,
        name: `${values.name} ${values.lastname}`.trim(),
        phone: values.phone,
        status: true,
        uid: userUid
      };
      
      await setDoc(doc(db, "u_users", userUid), finalData);
      toast.success("Registro exitoso.");
      navigate('/auth/login');
    }
  } catch (error) {
    console.error("Registration error:", error);
    toast.error(getErrorMessage(error.code));
    setSubmitting(false);
    setSubmit(false);
  }
}
```

## Forgot Password View

**Route:** `/auth/recovery`

### Purpose

Send password reset email to users who have forgotten their credentials.

### Implementation

```javascript
const ForgotPassword = () => {
  const RecoverySchema = Yup.object().shape({
    email: Yup.string()
      .email("El correo electrónico no es válido")
      .required("Se requiere correo electrónico")
  });

  const formik = useFormik({
    initialValues: {
      email: '',
    },
    validationSchema: RecoverySchema,
    onSubmit: async (values, { setSubmitting }) => {
      try {
        await sendPasswordResetEmail(auth, values.email);
        toast.success(
          "Se han enviado las instrucciones para recuperar tu contraseña a tu correo.",
          {
            position: "bottom-center",
            autoClose: 5000,
          }
        );
        setSubmitting(false);
      } catch (error) {
        toast.error(
          "Error al enviar el correo. Verifica que la dirección sea correcta.",
          {
            position: "bottom-center",
            autoClose: 5000,
          }
        );
        setSubmitting(false);
      }
    }
  });

  return (
    <Card className="w-full max-w-md">
      <CardHeader className="text-center">
        <CardTitle className="text-2xl font-bold">Recuperar contraseña</CardTitle>
        <CardDescription>
          Ingresa tu correo para recibir las instrucciones
        </CardDescription>
      </CardHeader>
      <Form onSubmit={formik.handleSubmit}>
        <CardContent>
          <div className="space-y-2">
            <Label htmlFor="email">Correo electrónico</Label>
            <Input
              id="email"
              placeholder="tu@ejemplo.com"
              type="email"
              {...formik.getFieldProps("email")}
            />
            {formik.touched.email && formik.errors.email && (
              <Alert variant="destructive">
                <AlertCircle className="h-4 w-4 mr-2" />
                {formik.errors.email}
              </Alert>
            )}
          </div>
          <Button
            type="submit"
            className="w-full mt-6 bg-primary hover:bg-primary/90 text-white"
            disabled={formik.isSubmitting}
          >
            {formik.isSubmitting ? 'Enviando...' : 'Enviar instrucciones'}
          </Button>
        </CardContent>
      </Form>
      <CardFooter className="flex flex-col space-y-4">
        <p className="text-sm text-gray-600">
          ¿Recordaste tu contraseña? <Link to="/auth/login" className="text-primary hover:underline">Volver al inicio de sesión</Link>
        </p>
        <p className="text-sm text-gray-600">
          ¿No tienes una cuenta? <Link to="/auth/register" className="text-primary hover:underline">Regístrate</Link>
        </p>
      </CardFooter>
    </Card>
  );
};
```
