## Overview

The User Profile view (`src/views/user/profile/ProfileUser.js`) provides authenticated users with access to their account information, purchased tickets, and preferences.

**Route:** `/profileUser`  
**Guard:** `<AuthGuard>` - Requires authentication

## Ticket Management Tab

### Fetching User Tickets

```javascript
useEffect(() => {
  const fetchUserTickets = async () => {
    if (!authUser) return;

    setLoading(true);
    try {
      const ticketsCollectionRef = collection(db, `u_users/${authUser.uid}/tickets`);
      const ticketsSnapshot = await getDocs(ticketsCollectionRef);

      const ticketsData = [];
      ticketsSnapshot.forEach((doc) => {
        ticketsData.push({
          id: doc.id,
          ...doc.data()
        });
      });

      setUserTickets(ticketsData);
    } catch (error) {
      console.error("Error al obtener tickets del usuario:", error);
    } finally {
      setLoading(false);
    }
  };

  fetchUserTickets();
}, [authUser]);
```

### Ticket Card Component

```javascript
const TicketCard = ({ ticket }) => {
  const formatFirestoreDate = (firestoreDate) => {
    if (!firestoreDate) return "Fecha no disponible";
    
    if (firestoreDate.seconds) {
      return new Date(firestoreDate.seconds * 1000).toLocaleString();
    } else if (typeof firestoreDate === 'string') {
      return new Date(firestoreDate).toLocaleString();
    }
    
    return "Fecha no disponible";
  };
  
  const getTicketStatus = (status, ledger) => {
    if (ledger && ledger.length > 0) {
      const lastAction = ledger[ledger.length - 1].action;
      
      switch (lastAction) {
        case "generated":
          return { text: "Generado", color: "bg-blue-100 text-blue-800" };
        case "sold":
          return { text: "Vendido", color: "bg-green-100 text-green-800" };
        case "used":
          return { text: "Usado", color: "bg-gray-100 text-gray-800" };
        case "cancelled":
          return { text: "Cancelado", color: "bg-red-100 text-red-800" };
        default:
          return { text: "Pendiente", color: "bg-yellow-100 text-yellow-800" };
      }
    }
    
    return status 
      ? { text: "Activo", color: "bg-green-100 text-green-800" }
      : { text: "Inactivo", color: "bg-red-100 text-red-800" };
  };
  
  const ticketStatus = getTicketStatus(ticket.status, ticket.ledger);

  return (
    <Card className="flex flex-col md:flex-row overflow-hidden rounded-xl hover:shadow-lg transition-shadow border-l-4" 
          style={{ borderLeftColor: ticket.color || '#0693e3' }}>
      <div className="w-full md:w-1/3 bg-gradient-to-r from-primary to-secondary p-4 flex flex-col justify-center items-center text-white">
        <div className="text-2xl font-bold mb-2">{ticket.event_name}</div>
        <div className="text-sm opacity-90 mb-4">{ticket.event_description}</div>
        <div className="bg-white/20 rounded-lg p-3 w-full">
          <div className="text-xs mb-1">Fecha y hora:</div>
          <div className="font-medium">{formatFirestoreDate(ticket.date_start)}</div>
        </div>
      </div>
      
      <CardContent className="flex-1 p-4">
        <div className="flex justify-between items-start mb-4">
          <div>
            <CardTitle className="text-xl font-semibold">{ticket.event_name}</CardTitle>
            <p className="text-gray-600 text-sm">{ticket.event_description}</p>
          </div>
          <div className={`text-xs font-medium rounded-full px-3 py-1 ${ticketStatus.color}`}>
            {ticketStatus.text}
          </div>
        </div>
        
        <div className="grid grid-cols-1 md:grid-cols-2 gap-3 mb-4">
          <div>
            <div className="text-xs text-gray-500">Zona</div>
            <div className="font-medium">{ticket.zone || "No especificada"}</div>
          </div>
          <div>
            <div className="text-xs text-gray-500">Asiento</div>
            <div className="font-medium">{ticket.seat_id || "No asignado"} ({ticket.seat_row || "Fila no asignada"})</div>
          </div>
          <div>
            <div className="text-xs text-gray-500">Cliente</div>
            <div className="font-medium">{ticket.customer_name}</div>
          </div>
          <div>
            <div className="text-xs text-gray-500">Contacto</div>
            <div className="font-medium">{ticket.customer_phone}</div>
          </div>
        </div>
        
        <div className="text-xs text-gray-500 mb-1">Historial</div>
        <div className="bg-gray-50 p-2 rounded text-xs space-y-1">
          {ticket.ledger && ticket.ledger.map((entry, index) => (
            <div key={index} className="flex justify-between">
              <span className="font-medium">{entry.action}</span>
              <span>{formatFirestoreDate(entry.date)}</span>
            </div>
          ))}
        </div>
      </CardContent>
      
      <CardFooter className="p-4 flex flex-col gap-2 md:border-l">
        <Button variant="outline" className="w-full">
          <Download className="h-4 w-4 mr-2" />
          Descargar
        </Button>
        <Button className="w-full bg-primary hover:bg-primary/90 text-white">
          <Ticket className="h-4 w-4 mr-2" />
          Ver Detalle
        </Button>
      </CardFooter>
    </Card>
  );
};
```
