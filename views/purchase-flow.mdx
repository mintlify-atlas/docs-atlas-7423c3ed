## Overview

The purchase flow consists of two main views:
- **PurchaseOrder** (`src/views/purchaseOrder/PurchaseOrder.js`) - Ticket selection and payment
- **PurchaseSuccess** (`src/views/purchaseSuccess/PurchaseSuccess.js`) - Order confirmation

## Purchase Order View

**Route:** `/purchaseOrder/:idEvent`  
**Guard:** `<AuthGuard>` - Requires authentication

### Purpose

Guides users through a multi-step process to select seats, enter buyer information, and complete payment for event tickets.

```javascript
const generateSeats = (zone, tickets) => {
  if (zone.type === "Libre Asignación") {
    const availableTickets = tickets.filter(t => t.status === true && t.status_offline !== true);
    return {
      type: "Libre Asignación",
      availableSeats: availableTickets.length,
      price: zone.price,
      selectedCount: 0,
      tickets: availableTickets
    };
  }

  const sortedTickets = tickets.sort((a, b) => {
    const numA = parseInt(a.seat_id.match(/\d+/)[0], 10);
    const numB = parseInt(b.seat_id.match(/\d+/)[0], 10);
    return numA - numB;
  });

  return {
    type: "Asignado",
    seats: sortedTickets.map((ticket) => ({
      id: ticket.seat_id,
      ticket_id: ticket.ticket_id,
      price: zone.price,
      status: ticket.status === true ? 'available' : 'unavailable',
      status_offline: ticket.status_offline,
      location: zone.name,
      name: zone.name,
      metadata: ''
    }))
  };
};
```

## Price Calculation

### Tax and Fee Computation

```javascript
const truncateDecimals = (number, digits) => {
  const multiplier = Math.pow(10, digits);
  return Math.trunc(number * multiplier) / multiplier;
};

const roundDecimals = (number, digits) => {
  const multiplier = Math.pow(10, digits);
  return Math.round(number * multiplier) / multiplier;
};

// Calculate totals
const subtotal = selectedSeats.reduce((acc, seat) => acc + seat.price, 0);
const ivaAmount = roundDecimals(subtotal * 0.16, 2);
const igtfAmount = paymentMethod === 'Zelle' ? truncateDecimals((subtotal + ivaAmount) * 0.03, 2) : 0;
const totalAmount = subtotal + ivaAmount + igtfAmount;
```

## Order Creation

When payment is confirmed:

```javascript
const handleCompletePurchase = async () => {
  setSubmit(true);
  
  try {
    const orderData = {
      event_id: params.idEvent,
      tickets: selectedSeats.map(seat => ({
        ticket_id: seat.ticket_id,
        seat_id: seat.id,
        zone: seat.location,
        amount: seat.price
      })),
      purchaser_info: buyerInfo,
      recipient_info: isGiftPurchase ? recipientInfo : null,
      is_gift: isGiftPurchase,
      amount: totalAmount,
      exchange_rate: usdCentralBank,
      transactions: [{
        payment_name: paymentMethod,
        payment_data: paymentInfo,
        amount_currency: paymentMethod === 'Zelle' ? 'USD' : 'VEF'
      }]
    };
    
    const response = await order_created(orderData);
    
    if (response?.data?.success) {
      navigate(`/purchaseSuccess/${params.idEvent}/${response.data.order_id}`);
    }
  } catch (error) {
    console.error("Order creation error:", error);
    toast.error("Error al procesar la orden. Por favor, intenta nuevamente.");
  }
  
  setSubmit(false);
};
```

### Payment Details Rendering

Dynamically renders payment information based on method:

```javascript
const renderPaymentDetails = (data) => {
  const renderValue = (key, value) => {
    if (value === null || value === undefined) return "-";
    
    // Security: Hide CVV
    if (key === 'cvv') return '***';
    
    // Mask card number
    if (key === 'card_number' && typeof value === 'string') {
      if (value.length > 4) {
        return '•••• •••• •••• ' + value.slice(-4);
      }
      return value;
    }
    
    // Format transaction status
    if (key === 'transaction_status') {
      if (value === '1000') {
        return <span className="text-green-600 font-medium">Aprobada</span>;
      }
      return value;
    }
    
    // Format amounts
    if (key === 'transaction_amount') {
      return <span className="font-semibold text-green-600">Bs {parseFloat(value).toFixed(2)}</span>;
    }
    
    return value;
  };
  
  // Categorize fields into sections
  const payerFields = ['pagador_telefono', 'pagador_identificacion', 'origin_bank'];
  const recipientFields = ['destination_phone', 'destination_id', 'bank_name'];
  
  // Render sections
  return (
    <div>
      {/* Payer section */}
      {/* Recipient section */}
      {/* General fields */}
    </div>
  );
};
```
